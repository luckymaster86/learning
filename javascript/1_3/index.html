function voron(number){

	switch(number){
		case 1:return "<p class='voron'>На ветке одна ворона</p>";break;
		case 2:return "<p class='voron'>На ветке две вороны</p>";break;
		case 3:return "<p class='voron'>На ветке три вороны</p>";break;
		case 4:return "<p class='voron'>На ветке четыре вороны</p>";break;
		case 5:return "<p class='voron'>На ветке пять ворон</p>";break;
		case 6:return "<p class='voron'>На ветке шесть ворон</p>";break;
		case 7:return "<p class='voron'>На ветке семь ворон</p>";break;
		case 8:return "<p class='voron'>На ветке восемь ворон</p>";break;
		case 9:return "<p class='voron'>На ветке девять ворон</p>";break;
		case 10:return "<p class='voron'>На ветке десять ворон</p>";break;
		default:return "<p class='voron'>На ветке неизвестное количество ворон!</p>";
	}

}
document.write(voron(4));
</script>
